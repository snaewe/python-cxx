PyCXX - How to update from V5 to V6

  • Replace Py::Nothing with Py::None
  • Replace Py::Int with Py::Long
  • Replace Py::LongLong with Py::Long
  • Replace as_std_string() with as_std_string( "encoding" ) or as_std_string( NULL )
  • #if PY_MAJOR_VERSION >= 3
            src_url_or_path = py_src_url_or_path.as_std_string( name_utf8 );
#else
            if( py_src_url_or_path.isUnicode() )
            {
                Py::String utf8( py_src_url_or_path.encode( name_utf8 ) );
                src_url_or_path = py_src_url_or_path.as_std_string();
            }
#endif
